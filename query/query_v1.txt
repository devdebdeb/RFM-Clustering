SELECT
    odb.APPLICATION_USER_ID
    ,DATEDIFF(DAY, MAX(COALESCE(odb.CLOSED_AT, odb.UPDATED_AT, odb.CREATED_AT)::DATE), CURRENT_DATE) R
    ,COUNT(DISTINCT odb.ORDER_ID) F
    ,SUM(odb.GMV) M
FROM
    RP_GOLD_DB_PROD.DES_PROD.ORDER_DIMENSIONS_BR odb
WHERE 1=1
    AND odb.COUNT_TO_GMV
GROUP BY odb.APPLICATION_USER_ID
HAVING 1=1
    AND MIN(COALESCE(odb.CLOSED_AT, odb.UPDATED_AT, odb.CREATED_AT))::DATE <= DATEADD(MONTH, -6, CURRENT_DATE)
    AND COUNT(DISTINCT odb.ORDER_ID) > 2
    AND MAX(COALESCE(odb.CLOSED_AT, odb.UPDATED_AT, odb.CREATED_AT))::DATE >= DATEADD(MONTH, -6, CURRENT_DATE)
